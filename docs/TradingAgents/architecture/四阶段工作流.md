```python
# scheduler/workflow_scheduler.py

import asyncio
from langchain_openai import ChatOpenAI
from langchain_core.prompts import ChatPromptTemplate

async def run_trading_analysis(symbol: str, config: dict, progress_callback=None):
    """
    执行四阶段股票分析工作流
    """
    # 初始化状态和模型
    state = AnalysisState(symbol=symbol, user_id=config["user_id"])
    model = ChatOpenAI(model=config.get("model", "gpt-4o"))

    # === Phase 1: 并行分析师团队 ===
    if config.get("enable_phase1", True):
        phase1_tasks = [
            run_analyst(state, model, cfg)
            for cfg in config["phase1_analysts"]
        ]
        phase1_results = await asyncio.gather(*phase1_tasks)
        state.analyst_reports = phase1_results
        await push_progress("phase1_complete", phase1_results)

    # === Phase 2: 串行观点辩论 ===
    if config.get("enable_phase2", True):
        debate_results = []
        for turn in range(config.get("debate_rounds", 2)):
            bull = await run_debater(state, model, "bull")
            bear = await run_debater(state, model, "bear", bull)
            debate_results.extend([bull, bear])
        state.debate_turns = debate_results
        await push_progress("phase2_complete", debate_results)

    # === Phase 3: 并行风险评估 ===
    if config.get("enable_phase3", True):
        phase3_tasks = [
            run_risk_assessor(state, model, cfg)
            for cfg in config["phase3_assessors"]
        ]
        phase3_results = await asyncio.gather(*phase3_tasks)
        state.risk_assessments = phase3_results
        await push_progress("phase3_complete", phase3_results)

    # === Phase 4: 综合报告 ===
    if config.get("enable_phase4", True):
        final_report = await run_final_report(state, model)
        state.final_report = final_report
        await push_progress("phase4_complete", final_report)

    return state
```

## 二、Phase 1: 分析师团队

### 2.1 阶段目标

从多个维度对目标股票进行初步的信息收集和分析，为后续阶段提供原始素材和基础观点。

### 2.2 核心特性：完全动态化

**配置驱动架构**：
- 所有分析师节点都是动态创建的，从用户配置加载
- 智能体的数量、名称、角色定义、工具权限全部来自配置文件
- 用户可以添加、删除、修改 Phase 1 的分析师

**默认智能体配置**（4 个）：

| 智能体标识 | 名称 | 职责 | 常用工具 |
|------------|------|------|----------|
| `market_technical` | 市场技术分析师 | 关注价格走势、技术指标 | K线数据、技术指标计算 |
| `fundamentals_analyst` | 基本面分析师 | 关注财务数据、估值分析 | 财报数据、财务指标 |
| `sentiment_analyst` | 市场情绪分析师 | 关注资金流向、投资者情绪 | 资金流向、涨跌统计 |
| `news_analyst` | 新闻分析师 | 关注公司公告、行业新闻 | 公司公告、新闻资讯 |

### 2.3 并发控制

**并发度配置**：

- `max_concurrency`: 同时运行的智能体最大数量（默认：3）
- 通过配置文件中的 `phase1.max_concurrency` 设置
- 使用 `asyncio.Semaphore` 实现并发限制

### 2.4 实现示例

```python
# phases/phase1.py

import asyncio
from langchain_core.prompts import ChatPromptTemplate
from langchain_openai import ChatOpenAI

async def run_phase1_analysts(state: AnalysisState, model: ChatOpenAI, analyst_configs: list):
    """
    并行执行所有分析师

    使用 asyncio.gather() 实现真正的并行
    """
    async def run_single_analyst(config):
        # 创建提示词
        prompt = ChatPromptTemplate.from_messages([
            ("system", config["system_prompt"]),
            ("human", "请分析股票：{symbol}")
        ])

        # 如果配置了工具，绑定工具
        if config.get("tools"):
            model_with_tools = model.bind_tools(config["tools"])
        else:
            model_with_tools = model

        # 调用模型
        chain = prompt | model_with_tools
        result = await chain.ainvoke({"symbol": state.symbol})

        return {
            "analyst": config["name"],
            "report": result.content,
            "role": config["role"]
        }

    # 创建并发信号量
    semaphore = asyncio.Semaphore(analyst_configs[0].get("max_concurrency", 3))

    async def run_with_semaphore(config):
        async with semaphore:
            return await run_single_analyst(config)

    # 并行执行所有分析师
    tasks = [run_with_semaphore(config) for config in analyst_configs]
    results = await asyncio.gather(*tasks, return_exceptions=True)

    # 处理异常
    analyst_reports = []
    for i, result in enumerate(results):
        if isinstance(result, Exception):
            analyst_reports.append({
                "analyst": analyst_configs[i]["name"],
                "error": str(result)
            })
        else:
            analyst_reports.append(result)

    return analyst_reports
```

## 三、Phase 2: 观点辩论

### 3.1 阶段目标

通过设置对立观点进行深入辩论，挖掘潜在风险和被忽视的机会，并基于辩论结果形成具体的交易计划。

### 3.2 固定智能体结构

| 智能体标识 | 名称 | 角色 | 输入 | 输出 |
|------------|------|------|------|------|
| `bull_researcher` | 看涨辩手 | 寻找积极因素、增长机会 | analyst_reports | 看涨论据 |
| `bear_researcher` | 看跌辩手 | 寻找负面因素、估值风险 | analyst_reports + 看涨论据 | 看跌论据 |
| `research_manager` | 研究经理 | 综合正反双方观点 | all debate_turns | 辩论总结 |
| `trader` | 交易员 | 制定具体交易计划 | 辩论总结 | trading_plan |

### 3.3 实现示例

```python
# phases/phase2.py

import asyncio
from langchain_core.prompts import ChatPromptTemplate
from langchain_openai import ChatOpenAI

async def run_phase2_debate(state: AnalysisState, model: ChatOpenAI, debater_configs: list):
    """
    串行执行观点辩论

    看涨和看跌辩手交替发言，多轮辩论
    """
    debate_turns = []
    rounds = debater_configs[0].get("rounds", 2)

    for round_num in range(rounds):
        # 看涨辩手
        bull_prompt = ChatPromptTemplate.from_messages([
            ("system", debater_configs[0]["bull_prompt"]),
            ("human", "基于以下分析报告，阐述看涨观点：\n{reports}")
        ])
        bull_chain = bull_prompt | model
        bull_response = await bull_chain.ainvoke({
            "reports": format_reports(state.analyst_reports),
            "previous_turns": format_debate_turns(debate_turns)
        })

        debate_turns.append({
            "role": "bull",
            "content": bull_response.content,
            "round": round_num + 1
        })

        # 看跌辩手
        bear_prompt = ChatPromptTemplate.from_messages([
            ("system", debater_configs[0]["bear_prompt"]),
            ("human", "基于以下分析报告和看涨观点，阐述看跌观点：\n{reports}\n\n{bull_view}")
        ])
        bear_chain = bear_prompt | model
        bear_response = await bear_chain.ainvoke({
            "reports": format_reports(state.analyst_reports),
            "bull_view": bull_response.content
        })

        debate_turns.append({
            "role": "bear",
            "content": bear_response.content,
            "round": round_num + 1
        })

    return debate_turns
```

## 四、Phase 3: 风险评估

### 4.1 阶段目标

从多个风险偏好视角对拟议的交易计划进行独立评估，确保决策充分考虑了潜在风险。

### 4.2 固定智能体结构

| 智能体标识 | 名称 | 风险偏好 | 关注点 |
|------------|------|----------|--------|
| `aggressive_assessor` | 激进派评估师 | 高风险高收益 | 机会成本、踏空风险 |
| `conservative_assessor` | 保守派评估师 | 本金安全优先 | 下行保护、止损 |
| `neutral_assessor` | 中性派评估师 | 风险收益平衡 | VaR、最大回撤 |
| `chief_risk_officer` | 首席风控官 | 综合评估 | 风险等级、管理建议 |

### 4.3 实现示例

```python
# phases/phase3.py

import asyncio
from langchain_core.prompts import ChatPromptTemplate
from langchain_openai import ChatOpenAI

async def run_phase3_risk(state: AnalysisState, model: ChatOpenAI, assessor_configs: list):
    """
    并行执行风险评估
    """
    async def run_single_assessor(config):
        prompt = ChatPromptTemplate.from_messages([
            ("system", config["system_prompt"]),
            ("human", "基于以下辩论结果，从{perspective}角度评估风险：\n{debate}")
        ])
        chain = prompt | model
        result = await chain.ainvoke({
            "debate": format_debate_turns(state.debate_turns),
            "perspective": config["perspective"]
        })
        return {
            "assessor": config["name"],
            "assessment": result.content,
            "perspective": config["perspective"]
        }

    # 并行执行
    tasks = [run_single_assessor(config) for config in assessor_configs]
    results = await asyncio.gather(*tasks, return_exceptions=True)

    return [r for r in results if not isinstance(r, Exception)]
```

## 五、Phase 4: 综合报告

### 5.1 阶段目标

将所有阶段的输出进行汇总，生成结构化、易读的最终投资报告。

### 5.2 推荐等级定义

| 等级 | 代码 | 描述 | 适用场景 |
|------|------|------|----------|
| 强烈买入 | STRONG_BUY | 预期显著上涨，强烈推荐买入 | 多重利好、低估明显 |
| 买入 | BUY | 预期上涨，推荐买入 | 利好大于利空 |
| 持有 | HOLD | 维持现有仓位 | 涨跌不确定 |
| 卖出 | SELL | 预期下跌，建议卖出 | 利空大于利好 |
| 强烈卖出 | STRONG_SELL | 预期显著下跌，强烈建议卖出 | 多重利空、高估明显 |

### 5.3 实现示例

```python

    """

请基于以下分析结果，生成结构化的投资报告：

## Phase 1: 分析师团队报告
{analyst_reports}

## Phase 2: 观点辩论
{debate_turns}

## Phase 3: 风险评估

        "debate_turns": format_debate_turns(state.debate_turns),
        "risk_assessments": format_assessments(state.risk_assessments)
    })

    return result.content
```

## 六、调度器实现

### 6.1 完整调度器代码

```python
# scheduler/workflow_scheduler.py

import asyncio
from typing import Optional, Callable, Dict, Any, List
from pydantic import BaseModel

from langchain_openai import ChatOpenAI
from langchain_core.prompts import ChatPromptTemplate
from langchain_core.messages import AIMessage, HumanMessage

from ..phases.phase1 import run_phase1_analysts
from ..phases.phase2 import run_phase2_debate
from ..phases.phase3 import run_phase3_risk
from ..phases.phase4 import run_phase4_report


class AnalysisState(BaseModel):
    """分析状态模型"""
    user_id: str
    task_id: str
    stock_code: str

    # 阶段开关
    enable_phase1: bool = True
    enable_phase2: bool = True
    enable_phase3: bool = True
    enable_phase4: bool = True

    # 累积状态
    analyst_reports: List[Dict[str, Any]] = []
    debate_turns: List[Dict[str, Any]] = []
    risk_assessments: List[Dict[str, Any]] = []
    token_usage: List[Dict[str, Any]] = []
    tool_calls: List[Dict[str, Any]] = []
    errors: List[Dict[str, Any]] = []

    # 输出状态
    final_report: Optional[str] = None
    final_recommendation: Optional[str] = None
    buy_price: Optional[float] = None
    sell_price: Optional[float] = None
    status: str = "pending"

    # 进度追踪
    current_phase: int = 0
    progress: float = 0.0


async def run_trading_analysis(
    symbol: str,
    config: Dict[str, Any],
    progress_callback: Optional[Callable] = None
) -> AnalysisState:
    """
    执行四阶段股票分析工作流

    Args:
        symbol: 股票代码
        config: 分析配置
            - user_id: 用户 ID
            - task_id: 任务 ID
            - model: 模型名称（如 "gpt-4o"）
            - temperature: 温度参数
            - enable_phase1/2/3/4: 阶段开关
            - phase1_analysts: Phase 1 分析师配置列表
            - phase2_debaters: Phase 2 辩手配置
            - phase3_assessors: Phase 3 评估师配置
        progress_callback: 进度回调函数

    Returns:
        AnalysisState: 最终分析状态
    """
    # 初始化状态
    state = AnalysisState(
        stock_code=symbol,
        user_id=config.get("user_id"),
        task_id=config.get("task_id"),
        enable_phase1=config.get("enable_phase1", True),
        enable_phase2=config.get("enable_phase2", True),
        enable_phase3=config.get("enable_phase3", True),
        enable_phase4=config.get("enable_phase4", True)
    )

    # 初始化模型
    model = ChatOpenAI(
        model=config.get("model", "gpt-4o"),
        temperature=config.get("temperature", 0.7)
    )

    try:
        # === Phase 1: 并行分析师团队 ===
        if state.enable_phase1:
            state.current_phase = 1
            if progress_callback:
                await progress_callback("phase1_start", {"phase": 1, "symbol": symbol})

            phase1_results = await run_phase1_analysts(
                state, model, config.get("phase1_analysts", [])
            )
            state.analyst_reports = phase1_results
            state.progress = 0.25

            if progress_callback:
                await progress_callback("phase1_complete", {
                    "phase": 1,
                    "results": phase1_results
                })

        # === Phase 2: 串行观点辩论 ===
        if state.enable_phase2:
            state.current_phase = 2
            if progress_callback:
                await progress_callback("phase2_start", {"phase": 2})

            phase2_results = await run_phase2_debate(
                state, model, config.get("phase2_debaters", [])
            )
            state.debate_turns = phase2_results
            state.progress = 0.5

            if progress_callback:
                await progress_callback("phase2_complete", {
                    "phase": 2,
                    "results": phase2_results
                })

        # === Phase 3: 并行风险评估 ===
        if state.enable_phase3:
            state.current_phase = 3
            if progress_callback:
                await progress_callback("phase3_start", {"phase": 3})

            phase3_results = await run_phase3_risk(
                state, model, config.get("phase3_assessors", [])
            )
            state.risk_assessments = phase3_results
            state.progress = 0.75

            if progress_callback:
                await progress_callback("phase3_complete", {
                    "phase": 3,
                    "results": phase3_results
                })

        # === Phase 4: 综合报告 ===
        if state.enable_phase4:
            state.current_phase = 4
            if progress_callback:
                await progress_callback("phase4_start", {"phase": 4})

            final_report = await run_phase4_report(state, model)
            state.final_report = final_report
            state.progress = 1.0

            if progress_callback:
                await progress_callback("phase4_complete", {
                    "phase": 4,
                    "report": final_report
                })

        state.status = "completed"
        return state

    except Exception as e:
        state.status = "failed"
        state.errors.append({
            "phase": state.current_phase,
            "error": str(e)
        })
        if progress_callback:
            await progress_callback("error", {
                "phase": state.current_phase,
                "error": str(e)
            })
        raise
```

### 6.2 工具函数

```python
# scheduler/utils.py

def format_reports(reports: List[Dict[str, Any]]) -> str:
    """格式化分析师报告"""
    formatted = []
    for report in reports:
        formatted.append(f"## {report.get('analyst', 'Unknown')}\n{report.get('report', '')}")
    return "\n\n".join(formatted)


def format_debate_turns(turns: List[Dict[str, Any]]) -> str:
    """格式化辩论记录"""
    formatted = []
    for turn in turns:
        role = "看涨" if turn.get("role") == "bull" else "看跌"
        formatted.append(f"**{role}辩手 (第{turn.get('round', 1)}轮)**:\n{turn.get('content', '')}")
    return "\n\n".join(formatted)


def format_assessments(assessments: List[Dict[str, Any]]) -> str:
    """格式化风险评估"""
    formatted = []
    for assessment in assessments:
        formatted.append(f"## {assessment.get('assessor', 'Unknown')} ({assessment.get('perspective', 'N/A')})\n{assessment.get('assessment', '')}")
    return "\n\n".join(formatted)
```
